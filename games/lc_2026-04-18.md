**LC121 最多一次交易**：
- hold状态：要么之前hold，要么今天买
- cash状态：要么之前cash，要么今天卖

```python
def maxProfit(prices):
    cash, hold = 0, -prices[0]
    for p in prices[1:]:
        cash, hold = max(cash, hold + p), max(hold, -p)
    return cash
```

## 03:30 编辑距离专题

### LC72 编辑距离
```python
def minDistance(word1, word2):
    m, n = len(word1), len(word2)
    dp = [[0]*(n+1) for _ in range(m+1)]
    for i in range(m+1): dp[i][0] = i
    for j in range(n+1): dp[0][j] = j
    for i in range(1, m+1):
        for j in range(1, n+1):
            if word1[i-1] == word2[j-1]:
                dp[i][j] = dp[i-1][j-1]
            else:
                dp[i][j] = 1 + min(dp[i-1][j], dp[i][j-1], dp[i-1][j-1])
    return dp[m][n]
```

| "a" | "" | 1 | ✅ |

### 模式要点
- dp[i][j] = word1[:i]到word2[:j]的最小编辑距离
- 三操作：插入、删除、替换（代价都是1）
- 空间优化：可压缩到O(min(m,n))

### LC218 天际线问题
**核心**：扫描线算法
```python
import heapq
def getSkyline(buildings):
    # events: (x, -height, right) for left, (x, height, right) for right
    events = []
    for l, r, h in buildings:
        events.append((l, -h, r))
        events.append((r, h, r))
    events.sort()
    
    result, heap = [], [(0, float('inf'))]
    for x, neg_h, r in events:
        if neg_h < 0:  # left event
            if -neg_h > heap[0][0]:
                result.append([x, -neg_h])
            heapq.heappush(heap, (neg_h, r))
        else:  # right event
            # remove all buildings ending at this x
            while heap and heap[0][1] <= x:
                heapq.heappop(heap)
        # update height
        if -heap[0][0] != (result[-1][1] if result else 0):
            result.append([x, -heap[0][0]])
    return result
```

```python
# 期望: [2,10],[3,15],[7,12],[12,0],[15,10],[20,8],[24,0]
```

### LC785 判断二分图
```python
def isBipartite(graph):
    color = [-1] * len(graph)
    for start in range(len(graph)):
        if color[start] != -1: continue
        queue = [start]
        color[start] = 0
        while queue:
            node = queue.pop(0)
            for nei in graph[node]:
                if color[nei] == -1:
                    color[nei] = 1 - color[node]
                    queue.append(nei)
                elif color[nei] == color[node]:
                    return False
    return True
```

### LC1135 最低成本连线
```python
def minimumCost(n, connections):
    parent = list(range(n))
    def find(x):
        while parent[x] != x:
            parent[x] = parent[parent[x]]
            x = parent[x]
        return x
    def union(x, y):
        px, py = find(x), find(y)
        if px == py: return False
        parent[px] = py
        return True
    
    edges = sorted(connections, key=lambda x: x[2])
    cost, edges_used = 0, 0
    for x, y, w in edges:
        if union(x-1, y-1):
            cost += w
            edges_used += 1
            if edges_used == n-1: break
    return cost if edges_used == n-1 else -1
```

## 06:40 线段树进阶：区间更新+懒传播

### 线段树核心结构
```python
class SegTree:
    def __init__(self, nums):
        self.n = len(nums)
        self.size = 1
        while self.size < self.n: self.size <<= 1
        self.tree = [0] * (2 * self.size)
        self.lazy = [0] * (2 * self.size)
        # build
        for i in range(self.n):
            self.tree[self.size + i] = nums[i]
        for i in range(self.size - 1, 0, -1):
            self.tree[i] = self.tree[2*i] + self.tree[2*i+1]
    
    def _push(self, idx, left, right):
        """懒传播：把lazy值下推到子节点"""
        if self.lazy[idx] != 0 and idx < self.size:
            mid = (left + right) // 2
            # 累加到左子
            self.tree[2*idx] += self.lazy[idx] * (mid - left + 1)
            self.lazy[2*idx] += self.lazy[idx]
            # 累加到右子
            self.tree[2*idx+1] += self.lazy[idx] * (right - mid)
            self.lazy[2*idx+1] += self.lazy[idx]
            self.lazy[idx] = 0
    
    def _update(self, idx, left, right, ql, qr, val):
        if qr < left or right < ql: return
        if ql <= left and right <= qr:
            self.tree[idx] += val * (right - left + 1)
            self.lazy[idx] += val
            return
        self._push(idx, left, right)
        mid = (left + right) // 2
        self._update(2*idx, left, mid, ql, qr, val)
        self._update(2*idx+1, mid+1, right, ql, qr, val)
        self.tree[idx] = self.tree[2*idx] + self.tree[2*idx+1]
    
    def _query(self, idx, left, right, ql, qr):
        if qr < left or right < ql: return 0
        if ql <= left and right <= qr: return self.tree[idx]
        self._push(idx, left, right)
        mid = (left + right) // 2
        return self._query(2*idx, left, mid, ql, qr) + \
               self._query(2*idx+1, mid+1, right, ql, qr)
    
    def update(self, l, r, val): self._update(1, 0, self.size-1, l, r, val)
    def query(self, l, r): return self._query(1, 0, self.size-1, l, r)
```

## 06:40 线段树进阶：区间更新+懒传播

### 核心概念：懒传播（Lazy Propagation）
- **延迟更新**：不立即下推更新到叶子，只在需要时传播
- **核心字段**：tree[] 存真实值，lazy[] 存待下推增量
- **下推时机**：查询/更新子节点时，先 _push 懒值

### 模板：区间更新+区间查询
```python
class SegTree:
    def __init__(self, nums):
        self.size = 1
        while self.size < len(nums): self.size <<= 1
        self.tree = [0] * (2 * self.size)
        self.lazy = [0] * (2 * self.size)
        for i in range(len(nums)): self.tree[self.size + i] = nums[i]
        for i in range(self.size - 1, 0, -1):
            self.tree[i] = self.tree[2*i] + self.tree[2*i+1]

    def _push(self, idx, left, right):
        if self.lazy[idx] != 0 and idx < self.size:
            mid = (left + right) // 2
            self.tree[2*idx]     += self.lazy[idx] * (mid - left + 1)
            self.lazy[2*idx]     += self.lazy[idx]
            self.tree[2*idx+1]   += self.lazy[idx] * (right - mid)
            self.lazy[2*idx+1]   += self.lazy[idx]
            self.lazy[idx] = 0

    def _update(self, idx, left, right, ql, qr, val):
        if qr < left or right < ql: return
        if ql <= left and right <= qr:
            self.tree[idx] += val * (right - left + 1)
            self.lazy[idx] += val; return
        self._push(idx, left, right)
        mid = (left + right) // 2
        self._update(2*idx, left, mid, ql, qr, val)
        self._update(2*idx+1, mid+1, right, ql, qr, val)
        self.tree[idx] = self.tree[2*idx] + self.tree[2*idx+1]

    def _query(self, idx, left, right, ql, qr):
        if qr < left or right < ql: return 0
        if ql <= left and right <= qr: return self.tree[idx]
        self._push(idx, left, right)
        mid = (left + right) // 2
        return self._query(2*idx, left, mid, ql, qr) + \
               self._query(2*idx+1, mid+1, right, ql, qr)

    def update(self, l, r, val): self._update(1, 0, self.size-1, l, r, val)
    def query(self, l, r): return self._query(1, 0, self.size-1, l, r)
```

## 06:57 AC自动机：多模式串匹配

### 核心概念
- **前缀树(Trie)**：多个模式串共享前缀节点
- **Fail指针**：当前字符失配时跳转的最长后缀
- **Output传递**：fail链上任何节点是终态则当前也是终态

### 关键洞察：逆序Trie（LC1032字符流后缀）

## 17:11 单调栈专题

### 核心模板：递增栈计算矩形
```python
def largestRectangleArea(heights):
    stack, max_area = [], 0
    for i, h in enumerate(heights + [0]):  # 末尾0触发全部弹出
        while stack and heights[stack[-1]] > h:
            height = heights[stack.pop()]
            width = i if not stack else i - stack[-1] - 1
            max_area = max(max_area, height * width)
        stack.append(i)
    return max_area
```
