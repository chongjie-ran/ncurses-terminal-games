### AC自动机模式回顾
- 三要素：Trie + Fail指针 + BFS构建
- 匹配时：沿子节点走，否则沿fail指针回溯

### AC自动机模板（C++）
```cpp
struct Node {
    int next[26] = {-1};
    int fail = 0;
    int out = 0;  // 输出计数
};
vector<Node> ac(1);
void insert(string s) {
    int p = 0;
    for (char c : s) {
        int idx = c - 'a';
        if (ac[p].next[idx] == -1) {
            ac[p].next[idx] = ac.size();
            ac.emplace_back();
        }
        p = ac[p].next[idx];
    }
    ac[p].out++;
}
void build() {
    queue<int> q;
    for (int i = 0; i < 26; i++) {
        if (ac[0].next[i] != -1) {
            ac[ac[0].next[i]].fail = 0;
            q.push(ac[0].next[i]);
        } else {
            ac[0].next[i] = 0;
        }
    }
    while (!q.empty()) {
        int v = q.front(); q.pop();
        for (int i = 0; i < 26; i++) {
            int u = ac[v].next[i];
            if (u != -1) {
                ac[u].fail = ac[ac[v].fail].next[i];
                ac[u].out += ac[ac[u].fail].out;
                q.push(u);
            } else {
                ac[v].next[i] = ac[ac[v].fail].next[i];
            }
        }
    }
}
int query(string s) {
    int p = 0, res = 0;
    for (char c : s) {
        int idx = c - 'a';
        p = ac[p].next[idx];
        res += ac[p].out;
    }
    return res;
}
```
